Task 1


            /* 
                Є рядок "I am in the KPI". Зробити перші літери у кожному слові у верхньому регістрі. 
                Використовувати for або while
            */
            
            let str1 = "I am in the KPI";
            let words = str1.split(" ");
            for (let i = 0; i < words.length; i++) {
                words[i] = words[i][0].toUpperCase() + words[i].substring(1);
            }
            let result1 = words.join(" ");
            console.log(result1);  // Output: "I Am In The KPI"

Task 2


            /* 
                Є рядок "tseb eht ma i". Використовуючи цикли, зробити зворотний рядок, 
                (тобто остання літера стає першою, передостання - другою і.т.д.)
            */
            
            let str2 = "tseb eht ma i";
            let reversedStr2 = "";
            for (let i = str2.length - 1; i >= 0; i--) {
                reversedStr2 += str2[i];
            }
            console.log(reversedStr2);  // Output: "i am the best"

Task 3


            /* 
                Знайти факторіал числа – від 1 до n
                За допомогою циклів (без функції) обчислити факторіал числа 5.
            */
            
            let n = 5;
            let factorial = 1;
            for (let i = 1; i <= n; i++) {
                factorial *= i;
            }
            console.log(factorial);  // Output: 120

Task 4


            /* 
                Використовуючи цикли, створити рядок "Пишемо до 5і: 1, 2, 3, 4, 5"
            */
            
            let result4 = "Пишемо до 5і: ";
            for (let i = 1; i <= 5; i++) {
                result4 += i;
                if (i < 5) {
                    result4 += ", ";
                }
            }
            console.log(result4);  // Output: "Пишемо до 5і: 1, 2, 3, 4, 5"
